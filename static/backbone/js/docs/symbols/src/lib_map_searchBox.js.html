1 define([], function() {
  2 	/** 
  3      * Class that adds a Search Box to the map.
  4      * @class SearchBox
  5      * @param {google.maps.Map} map A google.maps.Map object, to which the
  6      * SearchBox should be attached.
  7      */
  8 	localground.map.controls.SearchBox = (function (map) {	
  9 		var searchBox = null;
 10 		var $input = $('<input class="controls address-input" \
 11 						   type="text" placeholder="Search for Places">');
 12 		
 13 		/** Creates an HTML search control, and attaches
 14 		 * it to the upper right-hand side of the map.
 15 		 */
 16 		var render = function() {
 17 			map.controls[google.maps.ControlPosition.RIGHT_TOP].push($input.get(0));
 18 			searchBox = new google.maps.places.SearchBox($input.get(0));
 19 			google.maps.event.addListener(searchBox, 'places_changed', function () {
 20 				search();
 21 			});
 22 			$input.keyup(function (event) {
 23 				if (event.keyCode == 13) {
 24 					search();
 25 				}
 26 			});
 27 		};
 28 		
 29 		/**
 30 		 * Performs the search based on the location that the user
 31 		 * entered into the search box. If a valid location is found,
 32 		 * this function also zooms the map to the resulting location.
 33 		 */
 34 		var search = function(){
 35 			var places = searchBox.getPlaces();
 36 			if (places) {
 37 				if (places.length == 0) return;
 38 				if (places[0].geometry.viewport) {
 39 					map.fitBounds(places[0].geometry.viewport);
 40 				}
 41 				else {
 42 					map.setCenter(places[0].geometry.location);
 43 					map.setZoom(17);
 44 				}
 45 			}
 46 		};
 47 		
 48 		//call render upon initialization
 49 		render();
 50 	});
 51 	return localground.map.controls.SearchBox;
 52 });
