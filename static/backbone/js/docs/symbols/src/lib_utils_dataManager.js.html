1 define(
  2 	[
  3 		"backbone",
  4 		"config",
  5 		"models/project",
  6 		"collections/projects",
  7 		
  8 	], function(Backbone, Config, Project, Projects) {
  9 	/**
 10 	 * The map DataManager class separates the temporary data
 11 	 * storage (data retrieved from various Local Ground Data
 12 	 * API queries) from the different map views that consume
 13 	 * this data.
 14 	 * @class DataManager
 15 	 */
 16 	localground.map.data.DataManager = function() {
 17 		/**
 18 		 * A dictionary of the various data types available (given
 19 		 * the projects that have been selected), and the corresponding
 20 		 * data records that have been pulled down from the Data API.
 21 		 */
 22 		this.collections = {};
 23 		/**
 24 		 * A list of all of the user's available projects
 25 		 */
 26 		this.projects = new Projects();
 27 		/**
 28 		 * The projects that are currently loaded in
 29 		 * corresponding map views.
 30 		 */
 31 		this.selectedProjects = new Projects();
 32 		
 33 		/**
 34 		 * Fetches the user's available projects from the data API.
 35 		 */
 36 		this.fetchProjects = function(){
 37 			this.projects.fetch({ reset: true, async: false})
 38 		};
 39 		
 40 		/**
 41 		 * Fetches a particular project from the data API.
 42 		 * @param {Integer} id
 43 		 * The id of the project of interest.
 44 		 */
 45 		this.fetchDataByProjectID = function(id) {
 46 			var that = this;
 47 			var project = new Project({id: id});
 48 			project.fetch({data: {format: 'json'}, success: function(r){
 49 				that.updateCollections(project);
 50 			}});
 51 		};
 52 		
 53 		/**
 54 		 * Removes project data from memory (which subsequently
 55 		 * removes this data from the views which are bound to
 56 		 * the data).
 57 		 * @param {Integer} id
 58 		 * The id of the project of interest.
 59 		 */
 60 		this.removeDataByProjectID = function(id){
 61 			//http://backbonejs.org/#Collection-remove
 62 			for (key in this.collections) {
 63 				var collection = this.collections[key];
 64 				var items = [];
 65 				collection.each(function(item) {
 66 					if (item.get("project_id") == id) {
 67 						items.push(item);
 68 					}
 69 				}, this);
 70 				
 71 				//remove items from the collection:
 72 				this.collections[key].remove(items);
 73 			}
 74 			
 75 			//remove selected project:
 76 			this.selectedProjects.remove({id: id});
 77 		};
 78 		
 79 		/**
 80 		 * Because projects have many different types of data
 81 		 * associated with them (which must all be treated slightly
 82 		 * differently), each type of data is locally stored in its
 83 		 * own collection. See config.js to view the various data
 84 		 * types associated with a particular project. The config
 85 		 * file coordinates between the data stored in the API, and
 86 		 * the backbone data structures that manipulate this data.
 87 		 * @param {String} key
 88 		 * The key refers to the object_type of the data of interest.
 89 		 */
 90 		this.getCollection = function(key) {
 91 			return this.collections[key];
 92 		};
 93 		
 94 		/**
 95 		 * Coordinates data pulled down from the data API
 96 		 * with the local Backbone collections being stored
 97 		 * and manipulated in memory.
 98 		 * @method updateCollections
 99 		 * @param {Project} project
100 		 * A project detail data structure returned from the
101 		 * Local Ground data API.
102 		 */
103 		this.updateCollections = function(project) {
104 			//add child data to the collection:
105 			var children = project.get("children");
106 			for (key in children) {
107 				var opts = Config[key.split("_")[0]];
108 				opts.name = children[key].name;
109 				var models = [];
110 				$.each(children[key].data, function(){
111 					models.push(new opts.Model(this));
112 				});
113 				updateCollection(key, models, opts);
114 			}
115 			
116 			//add new project to the collection:
117 			this.selectedProjects.add(project, {merge: true});
118 		};
119 		
120 		/**
121 		 * Helper function that files Backbone models into their
122 		 * appropriate Backbone collection
123 		 * @method updateCollections
124 		 * @param {String} key
125 		 * @param {Array} models
126 		 * A list of Backbone models
127 		 * @param {Object} opts
128 		 * An object that tells the function which collection
129 		 * type to instantiate, and the name of the collection
130 		 */
131 		var updateCollection = function(key, models, opts) {
132 			if (this.collections[key] == null) {
133 				this.collections[key] = new opts.Collection(models);
134 				if (key.indexOf("form") != -1) {
135 					this.collections[key].name = opts.name;
136 				}
137 			}
138 			this.collections[key].add(models, {merge: true});
139 		};
140 	};
141 	return localground.map.data.DataManager;
142 });
