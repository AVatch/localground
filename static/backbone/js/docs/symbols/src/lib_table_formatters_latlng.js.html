1 define([], function() {
  2 	/** 
  3      * Handles the updating of the geoJSON Point attribute of the
  4      * model, based on the lat/lng values. Basically, it coordinates
  5      * the updates across the two cells. A little hacky, but works
  6      * for now.
  7      * @class LatLngFormatter
  8      */
  9 	localground.table.LatLngFormatter = {
 10 		update: function (formattedData, model, key, index) {
 11 			//delete geometry:
 12 			if (formattedData == "") {
 13 				model.lat = null;
 14 				model.lng = null;
 15 				model.set("geometry", null);
 16 				return null;
 17 			}
 18 	
 19 			//set model attributes:
 20 			model[key] = parseFloat(formattedData) || null;
 21 			
 22 			//if it's null, it's an invalid floating point number:
 23 			if(model[key] == null) {
 24 				return undefined;
 25 			}
 26 			
 27 			// update exisiting geometry (if exists):
 28 			geom = model.get("geometry");
 29 			if (geom && geom != null) {
 30 				geom.coordinates[index] = model[key];
 31 				model.set("geometry", geom);
 32 				model.trigger('change', model);
 33 				return geom;
 34 			}
 35 			
 36 			// otherwise create a new geometry object
 37 			else if (model.lng != null && model.lat != null) {
 38 				geom = {
 39 					type: "Point",
 40 					coordinates: [
 41 						parseFloat(model.lng),
 42 						parseFloat(model.lat)
 43 					]
 44 				};
 45 				model.set("geometry", geom);
 46 				model.trigger('change', model);
 47 				return geom;
 48 			}
 49 			return null;
 50 		}
 51 	};
 52 	return localground.table.LatLngFormatter;
 53 });
