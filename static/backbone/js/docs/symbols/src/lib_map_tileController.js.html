1 define(["lib/map/tiles/mapbox", "lib/map/tiles/stamen"],
  2 	function() {
  3 	/** 
  4      * Class that controls the map's tile options.
  5      * @class TileController
  6      * @param {google.maps.Map} opts.map
  7      * A google.maps.Map object, to which the TileController
  8      * should be attached.
  9      * 
 10      * @param {Array} opts.overlays
 11      * A list of available overlays, retrieved from the Local Ground API.
 12      * 
 13      * @param {Integer} opts.activeMapTypeID
 14      * The tileset that should be initialized on startup.
 15      */
 16 	localground.map.controls.TileController = function(opts) {
 17 		var that = this;
 18 		/** Raw data array of map overlays, pulled from the Data API */
 19 		this.overlays = null;
 20 		var map = null;
 21 		var mapTypeIDs = [];
 22 		var typeLookup = {
 23 			stamen: localground.map.tiles.Stamen,
 24 			mapbox: localground.map.tiles.MapBox,
 25 		};
 26 		
 27 		/**
 28 		 * Gets the name of the tileset by the ID.
 29 		 * @method getMapTypeNamebyId
 30 		 * 
 31 		 * @param {Integer} id
 32 		 * The id of the tileset.
 33 		 */
 34 		this.getMapTypeNamebyId = function(id) {
 35 			var tileInfo = getTileInfo("id", id);
 36 			return tileInfo.name;
 37 		};
 38 		
 39 		/**
 40 		 * Gets the tile information by the id.
 41 		 * @method getMapTypeId
 42 		 * 
 43 		 */
 44 		this.getMapTypeId = function() {
 45 			var tileInfo = getTileInfo("name", map.getMapTypeId().toLowerCase());
 46 			return tileInfo.id;
 47 		};
 48 		
 49 		/**
 50 		* Sets the active basemap tileset on the map. Called
 51 		* from the HTML control.
 52 		* @method setActiveMapType
 53 		* 
 54 		* @param {Integer} id
 55 		* The id of the corresponding tileset.
 56 		*/
 57 		this.setActiveMapType = function(id) {
 58 			if (id == null) {
 59 				return;
 60 			}
 61 			var mapType = getTileInfo("id", id);
 62 			var sourceName = mapType.sourceName.toLowerCase();
 63 			if (sourceName == "google") {
 64 				map.setMapTypeId(mapType.providerID);
 65 			}
 66 			else {
 67 				map.setMapTypeId(mapType.name);
 68 			}
 69 		};
 70 		
 71 		
 72 		/**
 73 		 * Initializes the tilesets for the map and adds each
 74 		 * tileset option to the maptype control on the map.
 75 		 * @method
 76 		 * 
 77 		 */
 78 		var initTiles = function() {
 79 			//iterate through each of the user's basemap tilesets and add it to the map:
 80 			$.each(this.overlays, function() {
 81 				var sourceName = this.sourceName.toLowerCase()
 82 				if (sourceName == "stamen" || sourceName == "mapbox") {
 83 					MapType = typeLookup[sourceName];
 84 					mapTypeIDs.push(this.name);
 85 					map.mapTypes.set(
 86 						this.name,
 87 						new MapType({
 88 							styleID: this.providerID,
 89 							name: this.name,
 90 							max: this.max
 91 						})
 92 					);
 93 				}
 94 				else if (sourceName == "google") {
 95 					mapTypeIDs.unshift(this.providerID);
 96 				}
 97 				else {
 98 					alert("Error in localground.map.TileManager: unknown map type");
 99 				}
100 			});
101 			map.mapTypeControlOptions.mapTypeIds = mapTypeIDs;
102 		}
103 	
104 		/**
105 		 * @method
106 		 * Gets the tile information according to the key/value identifier.
107 		 */
108 		var getTileInfo = function(key, value){
109 			for(var i=0; i < this.overlays.length; i++) {
110 				if(value.toString().toLowerCase() == this.overlays[i][key].toString().toLowerCase()) {
111 					return this.overlays[i];
112 				}
113 			}
114 			return null;
115 		}
116 	
117 		/**
118 		 * Initializes the TileController
119 		 * @method initialize
120 		 *
121 		 * @param {google.maps.Map} opts.map
122 		 * A google.maps.Map object, to which the TileController
123      	 * should be attached.
124      	 * 
125      	 * @param {Array} opts.overlays
126      	 * A list of available overlays, retrieved from the Local Ground API.
127      	 * 
128      	 * @param {Integer} opts.activeMapTypeID
129      	 * The tileset that should be initialized on startup.
130      	 */
131 		var initialize = function(opts) {
132 			//initialize properties:
133 			this.overlays = opts.overlays;
134 			map = opts.map;
135 			
136 			//initialize tiles and set the active map type
137 			initTiles();
138 			that.setActiveMapType(opts.activeMapTypeID);
139 		};
140 	
141 		// call on initialization:
142 		initialize(opts);
143 	};
144 	
145 	
146 	return localground.map.controls.TileController;
147 });
