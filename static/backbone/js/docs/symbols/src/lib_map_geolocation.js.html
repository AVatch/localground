1 define([], function() {
  2 	/** 
  3      * Class that adds a Search Box to the map.
  4      * @class GeoLocation
  5      * @param opts Initialization options for the Geolocation class.
  6      * @param {google.maps.Map} opts.map A google.maps.Map object, to which the
  7      * GeoLocation control should reference.
  8      * @param {UserProfile} opts.userProfile A user's profile, which includes
  9      * his/her default location.
 10      * @param {DefaultLocation} opts.defaultLocation A default location,
 11      * which includes a zoom level (1 to 20) and a center point, which is a
 12      * google.maps.LatLng object.
 13      */
 14 	localground.map.controls.GeoLocation = (function (opts) {
 15 		var map;
 16 		try { map = opts.map; }
 17 		catch(ex) {}
 18 		if (map == null) {
 19 			alert("\"map\" option required for localground.map.GeoLocation object");
 20 			return;
 21 		}
 22 		var userProfile = opts.userProfile;
 23 		var defaultLocation = opts.defaultLocation;
 24 		
 25 		/** Initializes the geolocation object. If the userProfile's
 26 		 * location is defined, the map uses it. Otherwise, it tries to
 27 		 * use the browser's geolocation capabilities. And if this is
 28 		 * disallowed or unsupported, it uses the application's default
 29 		 * location setting. 
 30 		 */
 31 		var initGeolocation = function() {
 32 			if(userProfile) {
 33 				//alert("Go to preferred location");
 34 				map.setCenter(userProfile.center);
 35 				map.setZoom(userProfile.zoom);
 36 			}
 37 			else if(navigator.geolocation) {
 38 				//alert("Find current location");
 39 				var browserSupportFlag = true;
 40 				navigator.geolocation.getCurrentPosition(setPosition, handleNoGeolocation);
 41 			}
 42 			// Browser doesn't support Geolocation
 43 			else {
 44 				browserSupportFlag = false;
 45 				map.setCenter(defaultLocation.center);
 46 				map.setZoom(defaultLocation.zoom);
 47 				that.handleNoGeolocation(browserSupportFlag);
 48 			}
 49 		};
 50 		
 51 		/**
 52 		 * Used as a callback function from the Browser's geolocation API.
 53 		 */
 54 		var setPosition = function(position) {
 55 			var latLng = new google.maps.LatLng(
 56 								position.coords.latitude,
 57 								position.coords.longitude
 58 							);
 59 			map.setCenter(latLng);
 60 			map.setZoom(14);
 61 		};
 62 		
 63 		var handleNoGeolocation = function(browserSupportFlag) {
 64 			//do nothing
 65 		};
 66 		
 67 		//initialize the geolocation helper:
 68 		initGeolocation();
 69 	});
 70 	return localground.map.controls.GeoLocation;
 71 });
